  const a=new Date(d);a.setHours(0,0,0,0);
  const b=new Date(START_DATE);b.setHours(0,0,0,0);
  return Math.max(1,Math.floor((a-b)/86400000)+1);
}
function addDays(d,n){const dt=new Date(d);dt.setDate(dt.getDate()+n);return dt;}
function getWeekStr(d){
  const dt=new Date(d);dt.setHours(0,0,0,0);
  const day=dt.getDay();
  const mon=new Date(dt);mon.setDate(dt.getDate()-(day===0?6:day-1));
  return mon.toISOString().slice(0,10);
}

const MILESTONES={
  [dateStr(CET6_DATE)]:'📝 六级考试',
  [dateStr(EXAM_DATE)]:'🎯 考研初试',
  [dateStr(MAJOR_START)]:'📚 专业课启动',

function renderCal(){
  const vm=STATE.calViewDate?new Date(STATE.calViewDate):new Date();
  const y=vm.getFullYear(),mo=vm.getMonth();
  const first=new Date(y,mo,1).getDay();
  const days=new Date(y,mo+1,0).getDate();
  const offset=first===0?6:first-1;
  const today=new Date();today.setHours(0,0,0,0);
  const selectedStr=STATE.calViewDate||todayStr();
  const mNames=['一月','二月','三月','四月','五月','六月','七月','八月','九月','十月','十一月','十二月'];
  let html=`<div class="cal-month">
    <span class="cal-nav" onclick="calNav(-1)">‹</span>
    <span class="cal-month-nm">${y}年${mNames[mo]}</span>
    <span class="cal-nav" onclick="calNav(1)">›</span>
  </div>
  <div class="cal-grid">
    ${['一','二','三','四','五','六','日'].map(d=>`<div class="cal-wday">${d}</div>`).join('')}`;
  for(let i=0;i<offset;i++) html+='<div class="cal-cell empty"></div>';
  for(let d=1;d<=days;d++){
    const dt=new Date(y,mo,d);dt.setHours(0,0,0,0);
    const ds = dt.getFullYear() + '-' + 
           String(dt.getMonth()+1).padStart(2,'0') + '-' + 
           String(dt.getDate()).padStart(2,'0')
    const isToday=dt.getTime()===today.getTime();
    const isFuture=dt>today;
    const doneTasks=STATE.completedTasks[ds]||[];
    const allTasks=getDayTasks(dt);
    const allDone=allTasks.every(t=>doneTasks.includes(t.id));
    const partial=doneTasks.length>0&&!allDone;
    const isMile=!!MILESTONES[ds];
    const isSel=ds===selectedStr&&!isToday;
    let cls=isToday?'today':isMile?'milestone':allDone&&!isFuture?'done':partial&&!isFuture?'partial':isFuture?'future':'';
    if(isSel) cls+=' selected';
    const dot=doneTasks.length>0&&!isToday?'<div class="cal-dot"></div>':'';
    html+=`<div class="cal-cell ${cls}" onclick="calClick('${ds}')">${d}${dot}</div>`;
  }
  html+='</div>';
  document.getElementById('calWrap').innerHTML=html;
  // restore detail if date selected
  if(STATE.calViewDate) calClick(STATE.calViewDate,true);
}

function calNav(dir){
  const vm=STATE.calViewDate?new Date(STATE.calViewDate):new Date();
  vm.setDate(1);vm.setMonth(vm.getMonth()+dir);
  STATE.calViewDate=vm.toISOString().slice(0,10);
  renderCal();
}

function calClick(ds,silent){
  STATE.calViewDate=ds;
  if(!silent) renderCal();
  const dt=new Date(ds);dt.setHours(12,0,0,0);
  const today=new Date();today.setHours(0,0,0,0);
  const isFuture=dt>today;
  const isToday=ds===todayStr();
  const allTasks=getDayTasks(dt);
  const doneTasks=STATE.completedTasks[ds]||[];
  const isMile=MILESTONES[ds];
  let detailHtml=`<b>${dt.getMonth()+1}月${dt.getDate()}日</b>${isToday?' （今天）':''}<br><br>`;
  if(isMile) detailHtml+=`<span style="color:#E65100;font-weight:700">${isMile}</span><br><br>`;
  if(isFuture&&!isMile){
    detailHtml+='<span style="color:var(--t-light)">📅 计划任务：</span><br>'+
      allTasks.map(t=>`${t.ic} ${t.name}`).join('<br>');
  } else if(!isFuture){
    const completed=allTasks.filter(t=>doneTasks.includes(t.id));
    const missed=allTasks.filter(t=>!doneTasks.includes(t.id));
    if(completed.length>0)
      detailHtml+='✅ 已完成：<br>'+completed.map(t=>`${t.ic} ${t.name}`).join('<br>')+'<br>';
    if(missed.length>0){
      detailHtml+='<br>❌ 未完成：<br>'+missed.map(t=>
        `${t.ic} ${t.name} <button class="backfill-btn" onclick="backfillTask('${ds}','${t.id}',${t.reward})">补打卡</button>`
      ).join('<br>');
    }
    if(doneTasks.length===0&&!isToday)
      detailHtml+='<span style="color:var(--t-light)">该天未打卡</span><br>'+
        allTasks.map(t=>`${t.ic} ${t.name} <button class="backfill-btn" onclick="backfillTask('${ds}','${t.id}',${t.reward})">补打卡</button>`).join('<br>');
  }
  let det=document.getElementById('calDetail');
  if(!det){det=document.createElement('div');det.id='calDetail';det.className='cal-detail';}
  det.innerHTML=detailHtml;
  const calWrap=document.getElementById('calWrap');
  if(!document.getElementById('calDetail')) calWrap.after(det);
  else calWrap.parentNode.insertBefore(det,calWrap.nextSibling);
}

function getDayStreak(){
  let streak=0;
  const today=new Date();today.setHours(0,0,0,0);
  for(let i=0;i<365;i++){
    const d=new Date(today);d.setDate(today.getDate()-i);
    const ds=d.toISOString().slice(0,10);
    if((STATE.completedTasks[ds]||[]).length>0) streak++;
    else break;
  }
  return streak;
}

function timeAgo(ts){
  const d=Date.now()-ts;
  if(d<60000) return '刚刚';
  if(d<3600000) return Math.floor(d/60000)+'分钟前';
  if(d<86400000) return Math.floor(d/3600000)+'小时前';
  return Math.floor(d/86400000)+'天前';
}

function switchTab(name,el){
  ['chatView','studyView','shopView'].forEach(v=>
    document.getElementById(v).classList.add('hidden'));
  document.querySelectorAll('.tab').forEach(t=>t.classList.remove('active'));
  document.getElementById(name+'View').classList.remove('hidden');
  if(el) el.classList.add('active');
  else document.getElementById('tab-'+name)?.classList.add('active');
  const titles={chat:'江浔',study:'考研进程',shop:'海盗币商店'};
  document.getElementById('navTitle').innerText=titles[name]||'江浔';
  const ib=document.getElementById('inputBar');
  if(name==='chat') ib.classList.remove('hidden');
  else ib.classList.add('hidden');
  if(name==='study') renderStudyView();
  if(name==='shop'){renderShop();renderSettings();}
  closeDots();
}

function escHtml(s){
  if(!s)return'';
  return String(s).replace(/&/g,'&amp;').replace(/</g,'&lt;')
    .replace(/>/g,'&gt;').replace(/"/g,'&quot;');
}
function escAttr(s){
  if(!s)return'';
  return String(s).replace(/'/g,"\\'").replace(/"/g,'&quot;');
}
